the_dacke_feud = {
	potential = {
		tag = SWE
		normal_or_historical_nations = yes
		is_free_or_tributary_trigger = yes
		uses_doom = no
		NOT = { has_country_flag = had_the_dacke_feud }
		any_owned_province = {
			OR = {
				area = norrland_area
				area = svealand_area
				area = ostra_svealand_area
				area = gotaland_area
				area = vastra_gotaland_area
				area = skaneland_area
			}
		}
		NOT = { has_country_modifier = swe_support_of_the_peasantry }
	}


	can_start = {
		has_any_disaster = no
		is_religion_enabled = protestant
		NOT = { religious_unity = 0.9 }
		OR = {
			has_country_modifier = vasa_centralization_reforms
			NOT = { average_home_autonomy = 10 }
			is_year = 1550
		}
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			AND = {
				stability = 3
				average_home_autonomy = 75
				religious_unity = 1
			}
		}
	}

	
	
	progress = {
		modifier = {
			factor = 1
			any_owned_province = {
				NOT = {
					religion = catholic
				}
			}
		}
		modifier = {
			factor = 1
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 1
			war_exhaustion = 4
		}
		modifier = {
			factor = 1
			NOT = { average_home_autonomy = 10 }
		}
		modifier = {
			factor = 1
			NOT = { average_home_autonomy = 25 }
		}
		modifier = {
			factor = 1
			NOT = { religious_unity = 0.5 }
		}
		modifier = {
			factor = 1
			NOT = { religious_unity = 0.25 }
		}
	}




	
	can_end = {
		NOT = {
			has_spawned_rebels = particularist_rebels
		}
		NOT = {
			exists = SMA
		}
	}
	



	modifier = {
		global_unrest = 10
	}	
	
	
	on_start = secret_swedish_events.1
	on_end = the_dacke_feud.100
}
